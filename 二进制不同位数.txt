#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
int main ()
{
     int m = 0,n = 0;
     int arr1[32] = {0};
     int arr2[32] = {0};
     int i = 31;
     int count = 0;
     scanf ("%d %d",&m,&n);
     while((0 != m)||(0 != n))
     {
          arr1[i] = m%2;
          arr2[i] = n%2;
          m /= 2;
          n /= 2;
          i --;
     }
     for (i = 0; i< 32;i ++)
     {
          if(arr2[i] != arr1[i])
              count ++;
     }
     printf("有%d位不相同\n",count);
     return 0 ;
}
